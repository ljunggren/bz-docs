## Usage examples

This functionality is best descibes with an example

### Sample data for examples
```
{
  name:"name-1",
  value:"value-1",
  data:{
    name:"name-2",
    value:"value-2"
  },
  list:[
    {
      name:"name-3",
      value:"value-3"
    },
    {
      name:"name-4",
      value:"value-4",
      age:33
    },
    {
      obj:{
        name:"name-5",
        value:"value-5",
        age:33
      }
    }
  ]
}
```
### Example 1: Extracting a node
```
"name" ==>
{
  name:"name-1"
}
------------------------------------------------
"name(Name)" ==>
{
  Name:"name-1"
}
------------------------------------------------
"name()" ==>
"name-1"
------------------------------------------------
["name","value"] ==>
{
  name:"name-1",
  value:"value-1"
}
------------------------------------------------
"*.name" ==>
{
  name:"name-1",
  data:{
    name:"name-2"
  },
  list:[
    {
      name:"name-3"
    },
    {
      name:"name-4"
    },
    {
      obj:{
        name:"name-5"
      }
    }
  ]
}
------------------------------------------------
["*.name","*.value"] ==>
{
  name:"name-1",
  value:"value-1",
  data:{
    name:"name-2",
    value:"value-2"
  },
  list:[
    {
      name:"name-3"
    },
    {
      name:"name-4"
    },
    {
      name:"name-5"
    },
    {
      value:"value-3"
    },
    {
      value:"value-4"
    },
    {
      value:"value-5"
    }
  ]
}
------------------------------------------------
{"*":["name","value"]} ==>
{
  name:"name-1",
  value:"value-1",
  data:{
    name:"name-2",
    value:"value-2"
  },
  list:[
    {
      name:"name-3",
      value:"value-3"
    },
    {
      name:"name-4",
      value:"value-4",
      age:33
    },
    {
      obj:{
        name:"name-5",
        value:"value-5",
      }
    }
  ]
}
------------------------------------------------
"*().name" ==>
{
  name:[
    "name-1",
    "name-2",
    "name-3",
    "name-4",
    "name-5"
  ]
}
------------------------------------------------
"*(abc).name" ==>
{
  abc:{
    name:[
      "name-1",
      "name-2",
      "name-3",
      "name-4",
      "name-5"
    ]
  }
}
------------------------------------------------
"*(abc).*.name" ==>
{
  abc:{
    name:"name-1",
    data:{
      name:"name-2"
    },
    list:[
      {
        name:"name-3"
      },
      {
        name:"name-4"
      },

      }

  }

"data(abc).name" ==>
